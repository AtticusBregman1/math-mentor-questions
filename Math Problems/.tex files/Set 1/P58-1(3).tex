\noindent Find the measure of \angle \emph{x} and \angle \emph{y}.\footnote{Hosei University Girl's High School, Kanagawa}\\

\begin{center}
\section*{Solution}
\end{center}

\noindent \begin{math}
Answer: \angle x=35^\circ,\angle y=60^\circ
\end{math}\\

\noindent Proof (\(\angle x\)): \(\mathbf{\angle x=20^\circ+15^\circ=35^\circ}\).\\
\noindent Proof (\(\angle y\)): \(\mathbf{\angle y=20^\circ+25^\circ+15^\circ=60^\circ}\).
